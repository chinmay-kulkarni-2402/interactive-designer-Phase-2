<script>
(function(){
  const root=document.getElementById('json1-binding');
  const sel=root.querySelector('#userSelect');
  const bound=root.querySelector('#boundFields');
  const tHead=root.querySelector('#table1 thead');
  const tBody=root.querySelector('#table1 tbody');
  const jsonObj=(typeof jsonData1!=='undefined' && jsonData1.length)?jsonData1[0]:{};
  const users=jsonObj||{};
  const keys=Object.keys(users).filter(k=>/^user\d+$/i.test(k));
  function buildSelect(){
    sel.innerHTML=keys.map(k=>`<option value="${k}">${k}</option>`).join('')||`<option value="">(empty)</option>`;
  }
  function renderTable(user){
    const tbl=user&&user.table1?user.table1:null;
    if(!tbl){tHead.innerHTML='';tBody.innerHTML='';return;}
    const h=tbl.heading||{};
    const cols=Object.keys(h);
    tHead.innerHTML=cols.length?`<tr>${cols.map(c=>`<th>${h[c]}</th>`).join('')}</tr>`:'';
    const rows=tbl.data||[];
    tBody.innerHTML=rows.map(r=>`<tr>${cols.map(c=>`<td>${r[c]??''}</td>`).join('')}</tr>`).join('');
  }
  function applyToDom(userId){
    const obj=users[userId]||{};
    bound.querySelectorAll('[data-key]').forEach(el=>{
      const key=el.getAttribute('data-key');
      const mode=el.getAttribute('data-bind')||'input';
      const val=obj[key]??'';
      if(mode==='input'){ if(el.value!==String(val)) el.value=String(val); }
      else if(mode==='text'){ el.value=String(val); }
      else{ el.textContent=String(val); }
    });
    renderTable(obj);
  }
  function setUser(id){
    window.__currentUser=id;
    applyToDom(id);
  }
  function attachTwoWay(){
    bound.addEventListener('input',e=>{
      const el=e.target.closest('[data-key]');
      if(!el)return;
      const key=el.getAttribute('data-key');
      const id=window.__currentUser;
      if(!id||!key)return;
      if(!users[id])users[id]={};
      users[id][key]=el.value;
    });
  }
  buildSelect();
  const initial=keys[0]||'';
  if(initial){ sel.value=initial; setUser(initial); }
  sel.addEventListener('change',e=>setUser(e.target.value));
  attachTwoWay();
})();
</script>
